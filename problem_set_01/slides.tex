\author{Chiara Passaro}
\date{}

\begin{document}

\begin{frame}
\titlepage
\end{frame}

\begin{frame}{The Problem}
\begin{itemize}
\item Patients need kidney transplants
\item Many have willing donors
\item Donor and patient are often incompatible
\end{itemize}
\end{frame}

\begin{frame}{Why This Is a Market Design Problem}
\begin{itemize}
\item Payments for organs are illegal
\item Decentralized matching fails
\item Centralized design improves outcomes
\end{itemize}
\end{frame}

\begin{frame}{Kidney Exchange}
\begin{itemize}
\item Match incompatible donor–patient pairs
\item Exchanges occur in cycles
\item All surgeries happen simultaneously
\end{itemize}
\end{frame}

\begin{frame}{Graph Representation}
\begin{itemize}
\item Nodes = patient–donor pairs
\item Directed edges = compatibility
\item Cycles represent feasible trades
\end{itemize}
\end{frame}

\begin{frame}{Design Constraints}
\begin{itemize}
\item Medical feasibility
\item Incentive compatibility
\item Computational limits
\end{itemize}
\end{frame}

\begin{frame}{Altruistic Donors}
\begin{itemize}
\item Non-directed donors start chains
\item Chains increase efficiency
\item Not all transplants must be simultaneous
\end{itemize}
\end{frame}

\begin{frame}{Impact}
\begin{itemize}
\item Thousands of additional transplants
\item Implemented worldwide
\item Major success of market design
\end{itemize}
\end{frame}
